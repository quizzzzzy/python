class Father:
    def __init__(self):
        self.name = "отец"
        self.tools = []

    def weekend(self):
        print("заливаем бетон")

class Son(Father):
    def __init__(self):
        super()
        self.name = "сын"

    def weekend(self):
        print("копать огород")


father = Father()
son = Son()

father.weekend()
son.weekend()

print(father.tools)
print(son.tools)
